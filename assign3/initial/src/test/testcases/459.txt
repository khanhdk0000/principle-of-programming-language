Var: global;
		Function: factorial
		Parameter: x 
		Body:
		  If x == 0 Then
		    global = 1;
		    Return 1;
		  Else
		    Return x * factorial(x - 1);
		  EndIf.
		EndBody.

		Function: main
		Body:
		  Var: input, checkret, fact;
		  input = int_of_string(read());
		  fact = factorial(input);
		  checkret = global;
		  If checkret == 1 Then
		    print(string_of_int(fact));
		  Else
		    print("Error: invalid input.");
		  EndIf.
		  Return 0;
		EndBody.
